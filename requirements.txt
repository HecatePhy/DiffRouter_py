# DiffRouter Python dependencies.  Python 3.9, CUDA 12.x (reference: torch 2.8.0+cu128).
#
# !!! INSTALL TORCH FIRST, WITH CUDA !!!
# `pip install -r requirements.txt` alone pulls the CPU-only torch wheel from PyPI, so
# torch.cuda.is_available() is False and the global route silently runs on CPU (very
# slow, nvidia-smi shows nothing). Install the CUDA build that matches your driver BEFORE
# the rest -- find the driver's CUDA version in `nvidia-smi` (top-right) and pick cuXXX
# at or below it:
#     pip install torch --index-url https://download.pytorch.org/whl/cu121   # cu118/cu121/cu124...
# Verify: python -c "import torch; print(torch.cuda.is_available(), torch.version.cuda)"
# The torch line below is only a version floor; it does NOT guarantee a CUDA build.
torch>=2.0
numpy>=1.24

# Visualization — congestion maps in scripts/congestion_diag.py (optional; the script
# runs and prints numbers without it, only the PNG is skipped).
matplotlib>=3.5

# Wirelength metrics — Potter's wirelength_analyzer/wa.py and scripts/total_wirelength.py
# parse routed .phys via Cap'n Proto. Without this, CPWL and total-WL come out blank
# (the global route + guide export do NOT need it).
pycapnp>=1.3
